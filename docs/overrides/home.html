Easy to get started


#include "crow.h"

int main()
{
    crow::SimpleApp app;

    CROW_ROUTE(app, "/")([](){
        return "Hello world";
    });

    app.port(18080).run();
}

CROW_ROUTE(app, "/json")
([]{
    crow::json::wvalue x{{"({{"}}"message", "Hello, World!"{{"}});"}}
    x["message2"] = "Hello, World.. Again!";
    return x;
});

URL parameter support as well!

CROW_ROUTE(app,"/hello/<int>")
([](int count){
    return crow::response(std::to_string(count));
});
