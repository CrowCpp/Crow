272  /// Generate the websocket headers using an opcode and the message size (in bytes).
273  std::string build_header(int opcode, size_t size)
274  {
275  char buf[2 + 8] = "\x80\x00";
276  buf[0] += opcode;
277  if (size < 126)
278  {
279  buf[1] += static_cast<char>(size);
280  return {buf, buf + 2};
281  }
282  else if (size < 0x10000)
283  {
284  buf[1] += 126;
285  *(uint16_t*)(buf + 2) = htons(static_cast<uint16_t>(size));
286  return {buf, buf + 4};
287  }
288  else
289  {
290  buf[1] += 127;
291  *reinterpret_cast<uint64_t*>(buf + 2) = ((1 == htonl(1)) ? static_cast<uint64_t>(size) : (static_cast<uint64_t>(htonl((size)&0xFFFFFFFF)) << 32) | htonl(static_cast<uint64_t>(size) >> 32));
292  return {buf, buf + 10};
293  }

318 
319  ///
320  /// Involves:<br>
321  /// Handling headers (opcodes, size).<br>
322  /// Unmasking the payload.<br>
323  /// Reading the actual payload.<br>
324  void do_read()
325  {
326  if (has_sent_close_ && has_recv_close_)
327  {
328  close_connection_ = true;
329  adaptor_.shutdown_readwrite();
330  adaptor_.close();
331  check_destroy();
332  return;
333  }
334 
335  is_reading = true;
336  switch (state_)
337  {
338  case WebSocketReadState::MiniHeader:
339  {
340  mini_header_ = 0;
341  //asio::async_read(adaptor_.socket(), asio::buffer(&mini_header_, 1),
342  adaptor_.socket().async_read_some(
343  asio::buffer(&mini_header_, 2),
344  [this](const error_code& ec, std::size_t
345 #ifdef CROW_ENABLE_DEBUG
346  bytes_transferred
347 #endif
348  )
349 
350  {
351  is_reading = false;
352  mini_header_ = ntohs(mini_header_);
353 #ifdef CROW_ENABLE_DEBUG
354 
355  if (!ec && bytes_transferred != 2)
356  {
357  throw std::runtime_error("WebSocket:MiniHeader:async_read fail:asio bug?");
358  }
359 #endif
360 
361  if (!ec)
362  {
363  if ((mini_header_ & 0x80) == 0x80)
364  has_mask_ = true;
365  else //if the websocket specification is enforced and the message isn't masked, terminate the connection
366  {
367 #ifndef CROW_ENFORCE_WS_SPEC
368  has_mask_ = false;
369 #else
370  close_connection_ = true;
371  adaptor_.shutdown_readwrite();
372  adaptor_.close();
373  if (error_handler_)
374  error_handler_(*this, "Client connection not masked.");
375  check_destroy(CloseStatusCode::UnacceptableData);
376 #endif
377  }
378 
379  if ((mini_header_ & 0x7f) == 127)
380  {
381  state_ = WebSocketReadState::Len64;
382  }
383  else if ((mini_header_ & 0x7f) == 126)
384  {
385  state_ = WebSocketReadState::Len16;
386  }
387  else
388  {
389  remaining_length_ = mini_header_ & 0x7f;
390  state_ = WebSocketReadState::Mask;
391  }
392  do_read();
393  }
394  else
395  {
396  close_connection_ = true;
397  adaptor_.shutdown_readwrite();
398  adaptor_.close();
399  if (error_handler_)
400  error_handler_(*this, ec.message());
401  check_destroy();
402  }
403  });
404  }
405  break;
406  case WebSocketReadState::Len16:
407  {
408  remaining_length_ = 0;
409  remaining_length16_ = 0;

414  bytes_transferred

419  remaining_length_ = remaining_length16_;
420 #ifdef CROW_ENABLE_DEBUG
421  if (!ec && bytes_transferred != 2)
422  {
423  throw std::runtime_error("WebSocket:Len16:async_read fail:asio bug?");
424  }
425 #endif
426 
427  if (!ec)
428  {
429  state_ = WebSocketReadState::Mask;
430  do_read();
431  }
432  else
433  {
434  close_connection_ = true;

436  adaptor_.close();

438  error_handler_(*this, ec.message());
439  check_destroy();
440  }
441  });
442  }
443  break;
444  case WebSocketReadState::Len64:

579  /// Process the payload fragment.
580 
581  ///
582  /// Unmasks the fragment, checks the opcode, merges fragments into 1 message body, and calls the appropriate handler.
583  bool handle_fragment()
584  {
585  if (has_mask_)
586  {
587  for (decltype(fragment_.length()) i = 0; i < fragment_.length(); i++)
588  {
589  fragment_[i] ^= ((char*)&mask_)[i % 4];
590  }
591  }
592  switch (opcode())
593  {
594  case 0: // Continuation
595  {
596  message_ += fragment_;
597  if (is_FIN())
598  {
599  if (message_handler_)
600  message_handler_(*this, message_, is_binary_);
601  message_.clear();
602  }
603  }
604  break;
605  case 1: // Text
606  {
607  is_binary_ = false;
608  message_ += fragment_;
609  if (is_FIN())
610  {
611  if (message_handler_)
612  message_handler_(*this, message_, is_binary_);
613  message_.clear();
614  }
615  }
616  break;
617  case 2: // Binary
618  {
619  is_binary_ = true;
620  message_ += fragment_;
621  if (is_FIN())
622  {
623  if (message_handler_)
624  message_handler_(*this, message_, is_binary_);
625  message_.clear();
626  }
627  }
628  break;
629  case 0x8: // Close
630  {
631  has_recv_close_ = true;
632 
633 
634  uint16_t status_code = NoStatusCodePresent;
635  std::string::size_type message_start = 2;
636  if (fragment_.size() >= 2)
637  {
638  status_code = ntohs(((uint16_t*)fragment_.data())[0]);
639  } else {
640  // no message will crash substr
641  message_start = 0;
642  }
643 
644  if (!has_sent_close_)
645  {
646  close(fragment_.substr(message_start), status_code);
647  }
648  else
649  {
650 
651  close_connection_ = true;
652  if (!is_close_handler_called_)
653  {
654  if (close_handler_)
655  close_handler_(*this, fragment_.substr(message_start), status_code);
656  is_close_handler_called_ = true;
657  }
658  adaptor_.shutdown_readwrite();
659  adaptor_.close();
660 
661  // Close handler must have been called at this point so code does not matter
662  check_destroy();
663  return false;
664  }
665  }
666  break;
667  case 0x9: // Ping
668  {
669  send_pong(fragment_);
670  }
671  break;
672  case 0xA: // Pong
673  {
674  pong_received_ = true;
675  }
676  break;
677  }
678 
679  fragment_.clear();
680  return true;
681  }

683  /// Send the buffers' data through the socket.
684 
685  ///
686  /// Also destroys the object if the Close flag is set.
687  void do_write()
688  {
689  if (sending_buffers_.empty())
690  {
691  sending_buffers_.swap(write_buffers_);
692  std::vector<asio::const_buffer> buffers;
693  buffers.reserve(sending_buffers_.size());
694  for (auto& s : sending_buffers_)
695  {
696  buffers.emplace_back(asio::buffer(s));
697  }
698  auto watch = std::weak_ptr<void>{anchor_};
699  asio::async_write(
700  adaptor_.socket(), buffers,
701  [&, watch](const error_code& ec, std::size_t /*bytes_transferred*/) {
702  if (!ec && !close_connection_)
703  {
704  sending_buffers_.clear();
705  if (!write_buffers_.empty())
706  do_write();
707  if (has_sent_close_)
708  close_connection_ = true;
709  }
710  else
711  {
712  auto anchor = watch.lock();
713  if (anchor == nullptr) { return; }
714 
715  sending_buffers_.clear();
716  close_connection_ = true;
717  check_destroy();
718  }
719  });
720  }
721  }

723  /// Destroy the Connection.
724  void check_destroy(websocket::CloseStatusCode code = CloseStatusCode::ClosedAbnormally)
725  {
726  // Note that if the close handler was not yet called at this point we did not receive a close packet (or send one)
727  // and thus we use ClosedAbnormally unless instructed otherwise
728  if (!is_close_handler_called_)
729  if (close_handler_)
730  close_handler_(*this, "uncleanly", code);
731  handler_->remove_websocket(this);
732  if (sending_buffers_.empty() && !is_reading)
733  delete this;
734  }
